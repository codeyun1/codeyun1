## 가능한 분야

```javascript
let a;
a = "자바스크립트!"
console.log(a)

//console : 자바스크립트 !
```

```c
#include <"studio.h">

int main(void) {
   char[256] a = "c언어!"

   printf(`%s`, a)
}

/*
c언어
*/
```

```typescript
function say(text: string) {
   console.log(text)
}

say("타입 스크립트 !")

//console : 타입 스크립트 !
```

```python
a = "에이 귀찮은데 말하자면..."
b = "파이썬 !"

print(a)
print(b)

# "에이 귀찮은데 말하자면..."
# "파이썬 !"
```
